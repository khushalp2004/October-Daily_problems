
///3289. The Two Sneaky Numbers of Digitville
vector<int> getSneakyNumbers(vector<int>& nums) {
        unordered_map<int,int> freq;
        for(int num: nums){
            freq[num]++;
        }

        vector<int> result;

        for(auto &num: freq){
            if(num.second>=2){
                result.push_back(num.first);
            }
        }
        return result;
    }
