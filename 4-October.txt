

//11. Container With Most Water
//get maximum and min and find the volume of it
int maxArea(vector<int>& height) {
        int n=height.size();
        int ans=0;
        int start=0;
        int end=n-1;
        while(start<end){
            ans=max(ans,(end-start)*min(height[start],height[end]));
            if(height[start]<height[end]){
                start++;
            }else{
                end--;
            }
        }
        return ans;
    }
