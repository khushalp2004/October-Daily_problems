////3147. Taking Maximum Energy From the Mystic Dungeon
//starting from the end can reduce the complexity from O(n^2) to o(n*k);
int maximumEnergy(vector<int>& energy, int k) {
        int n=energy.size();
        int result=INT_MIN;
        for(int i=n-k;i<n;i++){
            int sum=0;
            for(int j=i;j>=0;j-=k){
                sum+=energy[j];
                result=max(result,sum);
            }
        }
        return result;
    }
