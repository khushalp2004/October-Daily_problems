
///2011. Final Value of Variable After Performing Operations


int finalValueAfterOperations(vector<string>& operations) {
        int n=operations.size();
        int init=0;
        for(int i=0;i<n;i++){
            if(operations[i]=="--X" || operations[i]=="X--"){
                init=init-1;
            }
            if(operations[i]=="++X" || operations[i]=="X++"){
                init=init+1;
            }
        }

        return init;
    }
